# Prefer `readonly` React props
**ID:** `JS-0498` | **Lien:** [DeepSource](https://deepsource.com/directory/javascript/issues/JS-0498)

![Minor](https://img.shields.io/badge/severity-minor-yellow) ![Bug Risk](https://img.shields.io/badge/type-bug_risk-green)

Requires that React props are marked as `$ReadOnly` as react props are immutable, and react will throw error when modifying a property. Marking prop shapes as `$ReadOnly` avoids these issues.

Whether a developer declares a component as a function or a class, it must never modify its own props. React is pretty flexible but it has a single strict rule: All React components must act like pure functions with respect to their props. Props are never to be updated.


## Bad Practice

```javascript
type Props = { }; class Card extends React.Component<Props> { }

type OtherProps = { link: string }; class Card extends React.Component<OtherProps> { }

class Card extends React.Component<{}> { }

type Props = { details: {} }; class Card extends React.Component<Props, State> { }

type Props = { }; class Card extends Component<Props> { }

type Props = { }; class Card extends PureComponent<Props> { }

export type Props = {}; class Card extends Component<Props> { }

type Props = {| link: string |}; class Card extends Component<Props> { }

type Props = {| +link: string, ...bar |}; class Card extends Component<Props> { }

type Props = {| +link: string, -details: number |}; class Card extends Component<Props> { }

type Props = { }; function Func(props: Props) { return <p /> }

type Props = { }; function Func(props: Props) { return cond ? <p /> : <span /> }

function Func(props: {}) { return <p /> }

export type Props = {}; function Func(props: Props) { return <p /> }
```

## Recommended

```javascript
class Card extends React.Component<$ReadOnly<{}>> { }

type Props = $ReadOnly<{}>; class Card extends React.Component<Props> { }

type Props = $ReadOnly<{}>; class Card extends React.PureComponent<Props> { }

class Card extends React.Component<$ReadOnly<{}, State>> { }

type Props = $ReadOnly<{}>; class Card extends React.Component<Props, State> { }

type Props = $ReadOnly<{}>; class Card extends Component<Props> { }

type Props = $ReadOnly<{}>; class Card extends PureComponent<Props> { }

type CityType = {}; class Card extends Bar<CityType> { }

class Card { }

export type Props = $ReadOnly<{}>; class Card extends Component<Props, State> { }

export type Props = $ReadOnly<{}>; export class Card extends Component<Props> { }

type Props = {| +link: string |}; class Card extends Component<Props> { }

type Props = {| +link: string, +details: number |}; class Card extends Component<Props> { }

type Props = $FlowFixMe; class Card extends Component<Props> { }

type Props = {||}; class Card extends Component<Props> { }

class Card extends Component<{||}> { }

class Card extends React.Component<UnknownProps> { }

import { type Props } from "file"; class Card extends React.Component<Props> { }

type Props = {}; function Func() { }

type Props = $ReadOnly<{}>; function Func(props: Props) { }

type Props = {}; function Func(props: OtherProps) { }

function Func() { return <p /> }

function Func(props: $FlowFixMe) { return <p /> }

function Func(props: {||}) { return <p /> }
```
